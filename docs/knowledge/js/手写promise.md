// resolvePromise 函数
function resolvePromise(promise2, x, resolve, reject) {
  if(x === promise2) {
    return reject(new TypeError('检测到了死循环链使用！'))
  }
  let called
  if(x !== null && (typeof x === 'object' || typeof x === 'function')) {
    try {
      let then = x.then
      if(typeof then === 'function') {
        then.call(
          x,
          y => {
            if(called) return
            called = true
            resolvePromise(promise2, y, resolve, reject)
          },
          err => {
            if(called) return
            called = true
            reject(err)
          }
        )
      } else {
        resolve(x)
      }
    }
    catch(err) {
      if(called) return
      called = true
      reject(err)
    }
  } else {
    resolve(x)
  }
}
